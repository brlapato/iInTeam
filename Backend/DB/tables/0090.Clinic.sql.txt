CREATE TABLE `Clinic` (
  `ClinicId` bigint NOT NULL AUTO_INCREMENT,
  `Name` varchar(255) DEFAULT NULL,
  `NameDetail` varchar(255) DEFAULT NULL,
  `Description` varchar(2500) DEFAULT NULL,
  `Sport` varchar(75) DEFAULT NULL,
  `StartDateTime` datetime NOT NULL,
  `EndDateTime` datetime NOT NULL,
  `OrgId` int DEFAULT NULL,
  `PlayerId` bigint NOT NULL,
  `Location` varchar(255) DEFAULT NULL,
  `LocationDetail` varchar(255) DEFAULT NULL,
  `HeadCoach` varchar(255) DEFAULT NULL,
  `AssistantCoach1` varchar(255) DEFAULT NULL,
  `AssistantCoach2` varchar(255) DEFAULT NULL,
  `AssistantCoach3` varchar(255) DEFAULT NULL,
  `AssistantCoach4` varchar(255) DEFAULT NULL,
  `PreClinicNotes` varchar(2500) DEFAULT NULL,
  `PostClinicNotes` varchar(2500) DEFAULT NULL,
  PRIMARY KEY (`ClinicId`),
  KEY `FK_OrgId` (`OrgId`),
  KEY `FKPlayerId` (`PlayerId`),
  CONSTRAINT `FK_Clinic_OrgId` FOREIGN KEY (`PlayerId`) REFERENCES `Player` (`PlayerId`),
  CONSTRAINT `FK_Clinic_PlayerId` FOREIGN KEY (`OrgId`) REFERENCES `Org` (`OrgId`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
